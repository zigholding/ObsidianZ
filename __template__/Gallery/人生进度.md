function datetime2date(t){
	let year = t.year.toString();
	let month = t.month.toString().padStart(2, '0');
	let day = t.day.toString().padStart(2, '0');
	let formattedTime = `${year}-${month}-${day}`;
	let t0 = new Date(formattedTime);
	return t0;
}
